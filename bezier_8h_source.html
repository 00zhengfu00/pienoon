21 // A cubic Bezier curve.
22 // Can be initialized with start and end positions and derivatives.
23 // Start is when x = 0. End is when x = 1. These values are chosen purposefully
24 // to keep the domain close to 0.
25 // Bezier curves are nice because they are generally well behaved in between
26 // the start and end points.
27 template<class Vector, class Scalar>
28 class BezierCurve {
29  public:
30  BezierCurve()
31  : a(static_cast<Vector>(0.0f)),
32  b(static_cast<Vector>(0.0f)),
33  c(static_cast<Vector>(0.0f)),
34  d(static_cast<Vector>(0.0f)),
35  start_x(static_cast<Scalar>(0.0f)),
36  one_over_width_x(static_cast<Scalar>(0.0f)) {
37  }
38 
39  void Initialize(const Vector& start_value, const Vector& start_derivative,
40  const Vector& end_value, const Vector& end_derivative,
41  Scalar start_x_param, Scalar end_x_param) {
42  // We scale the input so that start is x = 0, and end is x = 1.
43  // We want to find Bezier curve B(x) such that,
44  // B(0) = start_value,
45  // B'(0) = start_derivative,
46  // B(1) = end_value,
47  // B'(1) = end_derivative,
48  //
49  // B(x) = ax^3 + bx^2(1 - x) + cx(1 - x)^2 + d(1 - x)^3
50  // B'(x) = (3a - b)x^2 + 2(b - c)x(1 - x) + (c - 3d)(1 - x)^2
51  //
52  // Substituting 0 and 1 into the equations above gives,
53  // B(0) = d ==> d = start_value
54  // B(1) = a ==> a = end_value
55  // B'(0) = c - 3d ==> c = 3*start_value + start_derivative
56  // B'(1) = 3a - b ==> b = 3*end_value - end_derivative
57  //
58  a = end_value;
59  b = 3.0f * end_value - end_derivative;
60  c = 3.0f * start_value + start_derivative;
61  d = start_value;
62  start_x = start_x_param;
63  one_over_width_x = 1.0f / (end_x_param - start_x_param);
64  assert(one_over_width_x < std::numeric_limits<Scalar>::infinity());
65  }
66 
67  Vector Evaluate(Scalar unscaled_x) const {
68  const Scalar x = ScaleX(unscaled_x);
69 
70  // B(x) = ax^3 + bx^2(1 - x) + cx(1 - x)^2 + d(1 - x)^3
71  const Scalar one_minus_x = 1.0f - x;
72  const Scalar one_minus_x_squared = one_minus_x * one_minus_x;
73  const Scalar one_minus_x_cubed = one_minus_x_squared * one_minus_x;
74  const Scalar x_squared = x * x;
75  const Scalar x_cubed = x_squared * x;
76  return a * x_cubed +
77  b * x_squared * one_minus_x +
78  c * x * one_minus_x_squared +
79  d * one_minus_x_cubed;
80  }
81 
82  Vector Derivative(Scalar unscaled_x) const {
83  const Scalar x = ScaleX(unscaled_x);
84 
85  // B'(x) = 3at^2 + b(2t(1 - t) + -t^2) +
86  // c((1 - t)^2 + -2(1 - t)t) - 3d(1 - t)^2
87  // = (3a - b)x^2 + 2(b - c)(x(1 - x) + (c - 3d)(1 - x)^2
88  const Scalar one_minus_x = 1.0f - x;
89  return (3.0f * a - b) * x * x +
90  2.0f * (b - c) * x * one_minus_x +
91  (c - 3.0f * d) * one_minus_x * one_minus_x;
92  }
93 
94  Vector SecondDerivative(Scalar unscaled_x) const {
95  const Scalar x = ScaleX(unscaled_x);
96 
97  // B''(x) = 2(3a - b)x + 2(b - c)((1 - x) - x) - 2(c - 3d)(1 - x)
98  // = (6a - 2b)x + (2b - 2c)(1 - 2x) + (-2c + 6d)(1 - x)
99  // = (6a - 2b - 4b + 4c + 2c - 6d)x + (2b - 2c - 2c + 6d)
100  // = (6a - 6b + 6c - 6d)x + (2b - 4c + 6d)
101  // = 6(a - b + c - d)x + 2(b - 2c + 3d)
102  return 6.0f * (a - b + c - d) * x + 2.0f * (b - 2.0f * c + 3.0f * d);
103  }
104 
105  private:
106  Scalar ScaleX(Scalar unscaled_x) const {
107  return mathfu::Clamp<Scalar>(
108  (unscaled_x - start_x) * one_over_width_x,
109  static_cast<Scalar>(0.0f), static_cast<Scalar>(1.0f));
110  }
111 
112  // B(x) = ax^3 + bx^2(1 - x) + cx(1 - x)^2 + d(1 - x)^3
113  Vector a, b, c, d;
114 
115  // The valid range for x starts at start_x. X's before start_x get clamped.
116  Scalar start_x;
117 
118  // The valid range for x ends at start_x + 1/one_over_width_x.
119  // We record one-over the width to eliminate a floating point division.
120  Scalar one_over_width_x;
121 };
