32 // Entity Manager is the main piece of code that manages all entities and
33 // components in the game. Normally the game will instantiate one instance
34 // of this, and then use it to create and control entities. The main
35 // uses for this class are:
36 // * Creating/destroying new entities
37 // * Keeping track of all components
38 // * Spawning and populating new components from data
39 // * Updating entities/components. (Usually done once per frame.)
40 class EntityManager {
41  public:
42  EntityManager();
43  typedef VectorPool<Entity> EntityStorageContainer;
44 
45  // Helper function for marshalling data from a component.
46  // Returns nullptr if it couldn't find it for any reason.
47  template <typename T>
48  T* GetComponentData(EntityRef& entity) {
49  return static_cast<T*>(GetComponentDataAsVoid(
50  entity, ComponentIdLookup<T>::kComponentId));
51  }
52 
53  // Helper function for marshalling data from a component.
54  // Returns nullptr if it couldn't find it for any reason.
55  template <typename T>
56  const T* GetComponentData(const EntityRef& entity) const {
57  return static_cast<const T*>(GetComponentDataAsVoid(
58  entity, ComponentIdLookup<T>::kComponentId));
59  }
60 
61  // Helper function for getting a particular component, given its datatype.
62  // Asserts if it doesn't exist.
63  template <typename T>
64  T* GetComponent() {
65  ComponentId id = ComponentIdLookup<T>::kComponentId;
66  assert(id != kInvalidComponent);
67  assert(id < kMaxComponentCount);
68  return static_cast<T*>(components_[id]);
69  }
70 
71  // Helper function for getting a particular component, given its datatype.
72  // Asserts if it doesn't exist.
73  template <typename T>
74  const T* GetComponent() const {
75  ComponentId id = ComponentIdLookup<T>::kComponentId;
76  assert(id != kInvalidComponent);
77  assert(id < kMaxComponentCount);
78  return static_cast<const T*>(components_[id]);
79  }
80 
81  // Helper function for getting a particular component, given the component ID.
82  inline ComponentInterface* GetComponent(ComponentId id) {
83  assert(id < kMaxComponentCount);
84  return components_[id];
85  }
86 
87  // Helper function for getting a particular component, given the component ID.
88  // (Const version)
89  inline const ComponentInterface* GetComponent(ComponentId id) const {
90  assert(id < kMaxComponentCount);
91  return components_[id];
92  }
93 
94  // Allocates a new entity, which is registered with no components.
95  // Returns an entityref to the new entity.
96  EntityRef AllocateNewEntity();
97 
98  // Deletes an entity, removing it from our list, and clearing any component
99  // data associated with it.
100  // Note: Deletion is deferred until the end of the frame. If you want to
101  // delete something instantly, use DeleteEntityImmediately.
102  void DeleteEntity(EntityRef entity);
103 
104  // Deletes an entity instantly. In general, you should use DeleteEntity,
105  // (which defers deletion until the end of the update cycle) unless you have
106  // a very good reason for doing so.
107  void DeleteEntityImmediately(EntityRef entity);
108 
109  // Adds a new component to the entity manager. The id represents a unique
110  // identifier for that component, which will other code can use to
111  // reference that component.
112  template <typename T>
113  void RegisterComponent(ComponentInterface* new_component) {
114  RegisterComponentHelper(new_component, ComponentIdLookup<T>::kComponentId);
115  }
116 
117  // Removes all components from an entity, causing any associated components
118  // to drop any data they have allocated for the entity.
119  void RemoveAllComponents(EntityRef entity);
120 
121  // Iterates through all registered components, and causes them to update.
122  // delta_time represents the timestep since last update.
123  void UpdateComponents(WorldTime delta_time);
124 
125  // Clears all data from all components, then dumps the list of components
126  // themselves, and then dumps the list of entities. Basically resets
127  // the entity manager into its original state.
128  void Clear();
129 
130  // functions for returning iterators suitable for iterating over every
131  // active entity.
132  EntityStorageContainer::Iterator begin() { return entities_.begin(); }
133  EntityStorageContainer::Iterator end() { return entities_.end(); }
134 
135  // Registers a factory object to be used when entities are created from
136  // arbitrary data. The entity factory is responsible for taking random
137  // data, and correctly transforming it into an entity. (Usually used
138  // when loading entities from data files.)
139  void set_entity_factory(EntityFactoryInterface* entity_factory) {
140  entity_factory_ = entity_factory;
141  }
142 
143  // Creates an entity from raw data. Usually used by the entity factory
144  // specified in set_entity_factory.
145  EntityRef CreateEntityFromData(const void* data);
146 
147  // Registers an entity with a component. This causes the component to
148  // allocate data for the entity, and includes the entity in that component's
149  // update routines.
150  void AddEntityToComponent(EntityRef entity, ComponentId component_id);
151 
152  private:
153  // Does all the real work of registering a component, aside from template fun.
154  // In particular, verifies that the requested ID isn't already in use,
155  // puts a pointer to the new component in our components_ array, and
156  // sets starting variables and preforms initialization on the component.
157  void RegisterComponentHelper(ComponentInterface* new_component,
158  ComponentId id);
159 
160  // Given a component ID and an entity, returns all data associated
161  // with that entity from the given component. Will assert if the inputs
162  // are not valid. (entity is no longer active or component is invalid)
163  // Data is returned as a void pointer - the caller is expected to know how
164  // to interpret it.
165  void* GetComponentDataAsVoid(EntityRef entity, ComponentId component_id);
166 
167  // Given a component ID and an entity, returns all data associated
168  // with that entity from the given component. Will assert if the inputs
169  // are not valid. (entity is no longer active or component is invalid)
170  // Data is returned as a void pointer - the caller is expected to know how
171  // to interpret it.
172  const void* GetComponentDataAsVoid(EntityRef entity,
173  ComponentId component_id) const;
174 
175  // Delete all the entities we have marked for deletion.
176  void DeleteMarkedEntities();
177  // Storage of all the entities currently tracked by the entitymanager
178  EntityStorageContainer entities_;
179 
180  // All the components that are tracked by the system, and are ready to
181  // have entities added to them.
182  ComponentInterface* components_[kMaxComponentCount];
183  // Entities that we plan to delete at the end of the frame.
184  std::vector<EntityRef> entities_to_delete_;
185  // Factory used for spawning new entities from data. Provided by the
186  // calling program.
187  EntityFactoryInterface* entity_factory_;
188 };
