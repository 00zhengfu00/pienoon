23 namespace impel {

25 // Maps a unique 'id' to a data element. Keeps the data elements contiguous
26 // in memory, even when 'ids' are deleted. Only reallocates memory when a new
27 // highwater number of elements is reached.
28 template<class ImpelData>
29 class IdMap {
30  // Invalid array index.
31  typedef uint16_t DataIndex;
32  static const DataIndex kInvalidIndex = 0xFFFF;
33 
34  public:
35  // Accessors.
36  ImpelData& Data(ImpelId id) { return data_[Index(id)]; }
37  const ImpelData& Data(ImpelId id) const { return data_[Index(id)]; }
38  ImpelData* Begin() { return data_.data(); }
39  const ImpelData* Begin() const { return data_.data(); }
40  const ImpelData* End() const { return data_.data() + data_.size(); }
41  int Count() const { return static_cast<int>(data_.size()); }
42 
43  // Allocate data and associate a unique id to it. Note that the id may have
44  // been an id that was previously used and then Free'd.
45  ImpelId Allocate() {
46  // Allocate a spot at the end of data_.
47  const DataIndex index = static_cast<DataIndex>(data_.size());
48  data_.resize(index + 1);
49 
50  // Allocate an id. We try to recycle ids first to avoid growing
51  // id_to_index_.
52  ImpelId id = 0;
53  if (ids_to_recycle_.empty()) {
54  // Allocate a new id from the end of id_to_index_.
55  id = static_cast<ImpelId>(id_to_index_.size());
56  id_to_index_.resize(id + 1);
57 
58  } else {
59  // Grab id from the recycle list.
60  id = ids_to_recycle_.back();
61  ids_to_recycle_.pop_back();
62  }
63 
64  //　Map id to that spot.
65  id_to_index_[id] = index;
66  return id;
67  }
68 
69  // Free the data associated with 'id' by compacting the data_ array on top
70  // of it. Return 'id' to the list of eligable ids to allocate.
71  void Free(ImpelId id) {
72  // Plug hole in data_.
73  const DataIndex index = Index(id);
74  const DataIndex last_index = static_cast<DataIndex>(data_.size() - 1);
75  if (index != last_index) {
76  const DataIndex last_id = Id(last_index);
77  assert(last_id != static_cast<DataIndex>(kImpelIdInvalid));
78 
79  // Move last item in data_ to the index being deleted.
80  data_[index] = data_[last_index];
81 
82  // Remap id onto the index we just moved to.
83  id_to_index_[last_id] = index;
84  }
85 
86  // Remove the last item from data_. It's no longer being used.
87  data_.pop_back();
88 
89  // Mark the current id invalid.
90  id_to_index_[id] = kInvalidIndex;
91 
92  // Reuse this id so that the id_to_index_ map doesn't keep growing.
93  ids_to_recycle_.push_back(id);
94  }
95 
96  private:
97  // Returns the index corresponding to an id. Fast.
98  DataIndex Index(ImpelId id) const {
99  assert(0 <= id && id < static_cast<ImpelId>(id_to_index_.size()));
100  const DataIndex index = id_to_index_[id];
101  assert(index != kInvalidIndex);
102  return index;
103  }
104 
105  // Returns an id corresponding to an index. Slow. Should be called
106  // infrequently.
107  ImpelId Id(DataIndex index) const {
108  // Get id for the last index in data.
109  for (DataIndex id = 0; id < static_cast<DataIndex>(id_to_index_.size());
110  ++id) {
111  if (id_to_index_[id] == index)
112  return static_cast<ImpelId>(id);
113  }
114  return kImpelIdInvalid;
115  }
116 
117  // Map ImpelIds into the data_ array. Each id gets a unique index into data_.
118  // This map may have holes. That is id_to_index_[id] may be kInvalidIndex.
119  // When the map has a hole, that 'id' will be in ids_to_recycle_.
120  // Note that this is a vector (not a map) because it requires very quick
121  // access.
122  std::vector<DataIndex> id_to_index_;
123 
124  // An unordered collection of 'id's that can be reused. We try to reuse ids
125  // so that the id_to_index_ map doesn't grow without bound.
126  std::vector<ImpelId> ids_to_recycle_;
127 
128  // A packed array of (template defined) data. There are no holes in this
129  // data. No holes allows for good memory cohesion and possible optimizations.
130  std::vector<ImpelData> data_;
131 };
132 
133 } // namespace impel
