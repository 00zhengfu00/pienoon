42 // Used to record state for fingers, mousebuttons, keys and gamepad buttons.
43 // Allows you to know if a button went up/down this frame.
44 class Button {
45  public:
46  Button() : is_down_(false) { AdvanceFrame(); }
47  void AdvanceFrame() { went_down_ = went_up_ = false; }
48  void Update(bool down);
49 
50  bool is_down() const { return is_down_; }
51  bool went_down() const { return went_down_; }
52  bool went_up() const { return went_up_; }
53 
54  private:
55  bool is_down_;
56  bool went_down_;
57  bool went_up_;
58 };

75 
76  SDLK_PAD_UP = -20,
77  SDLK_PAD_DOWN,
