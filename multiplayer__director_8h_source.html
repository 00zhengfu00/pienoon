33 // MultiplayerDirector is used for the multiscreen version of Pie Noon.
34 //
35 // It is responsible for managing the timings of all of the turns, receiving the
36 // messages from clients saying what actions they wish to perform, and for the
37 // formatting and sending of multiplayer messages between the host and client.
38 //
39 // When you create MultiplayerDirector, you must register multiple
40 // MultiplayerControllers (one for each player), which is how
41 // MultiplayerDirector can direct what the characters do each turn.
42 class MultiplayerDirector {
43  public:
44  MultiplayerDirector();
45 
46  // Give the multiplayer director everything it will need.
47  void Initialize(GameState *gamestate_ptr, const Config *config);
48 #ifdef PIE_NOON_USES_GOOGLE_PLAY_GAMES
49  // Register a pointer to GPGMultiplayer, so we can send multiplayer messages.
50  void RegisterGPGMultiplayer(GPGMultiplayer *gpg_multiplayer) {
51  gpg_multiplayer_ = gpg_multiplayer;
52  }
53 #endif
54  // Register one MultiplayerController assigned to each player.
55  void RegisterController(MultiplayerController *);
56 
57  // Call this each frame if multiplayer gameplay is going on.
58  void AdvanceFrame(WorldTime delta_time);
59 
60  // Start a new multi-screen game.
61  void StartGame();
62 
63  // End the multi-screen game.
64  void EndGame();
65 
66  // If testing on PC, pass in your PC keyboard input system to use debug
67  // keys for testing turn-based timings.
68  void SetDebugInputSystem(InputSystem *input) { debug_input_system_ = input; }
69 
70 #ifdef PIE_NOON_USES_GOOGLE_PLAY_GAMES
71  // Tell one of your connected players what his player number is.
72  void SendPlayerAssignmentMsg(const std::string &instance, CharacterId id);
73  // Broadcast start-of-turn to the players.
74  void SendStartTurnMsg(unsigned int turn_seconds);
75  // Broadcast end-of-game message to the players.
76  void SendEndGameMsg();
77  // Broadcast player health to the players.
78  void SendPlayerStatusMsg();
79 #endif
80 
81  // Takes effect when the next turn starts.
82  void set_seconds_per_turn(unsigned int seconds) {
83  seconds_per_turn_ = seconds;
84  }
85 
86  unsigned int seconds_per_turn() { return seconds_per_turn_; }
87 
88  // First turn is numbered 1, second turn numbered 2, etc.
89  // Is 0 before the first turn starts.
90  unsigned int turn_number() { return turn_number_; }
91 
92  // Tell the multiplayer director about a player's input.
93  void InputPlayerCommand(CharacterId id,
94  const multiplayer::PlayerCommand &command);
95 
96  // Internally, call this when a player has been hit by a pie. The multiplayer
97  // director will decide whether that player should be "stunned" by the hit
98  // and have one or more of his buttons locked for a turn.
99  void TriggerPlayerHitByPie(CharacterId player, int damage);
100 
101  // Is this an AI player or a human player?
102  bool IsAIPlayer(CharacterId player);
103 
104  // How long until the current turn ends? 0 if outside a turn.
105  WorldTime turn_timer() { return turn_timer_; }
106  // How long until the next turn starts? 0 if in a turn.
107  WorldTime start_turn_timer() { return start_turn_timer_; }
108 
109  // Set the number of AI players. The last N players are AIs.
110  void set_num_ai_players(unsigned int n) { num_ai_players_ = n; }
111  unsigned int num_ai_players() const { return num_ai_players_; }
112 
113  private:
114  struct Command {
115  CharacterId aim_at;
116  bool is_firing;
117  bool is_blocking;
118  Command() : aim_at(-1), is_firing(false), is_blocking(false) {}
119  };
120 
121  void TriggerStartOfTurn();
122  void TriggerEndOfTurn();
123  unsigned int CalculateSecondsPerTurn(unsigned int turn_number);
124 
125  // Get all the players' healths so we can send them in an update
126  std::vector<uint8_t> ReadPlayerHealth();
127 
128  // Tell the multiplayer director to choose AI commands for this player.
129  void ChooseAICommand(CharacterId id);
130 
131  void DebugInput(InputSystem *input);
132 
133  // Get all the players' onscreen splats to send in an update
134  std::vector<uint8_t> ReadPlayerSplats();
135 
136  GameState *gamestate_; // Pointer to the gamestate object
137  const Config *config_; // Pointer to the config structure
138 
139  std::vector<MultiplayerController *> controllers_;
140  std::vector<uint8_t> character_splats_;
141  // How long the current turn lasts.
142  WorldTime turn_timer_;
143  // In how long to start the next turn.
144  WorldTime start_turn_timer_;
145  unsigned int seconds_per_turn_;
146  unsigned int turn_number_;
147 
148  unsigned int num_ai_players_; // the last N players are AI.
149 
150  InputSystem *debug_input_system_;
151 
152  std::vector<Command> commands_;
153 
154 #ifdef PIE_NOON_USES_GOOGLE_PLAY_GAMES
155  GPGMultiplayer *gpg_multiplayer_ = nullptr;
156 #endif
157 
158  bool game_running_;
159 };
