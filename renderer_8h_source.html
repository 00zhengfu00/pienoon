29 // The core of the rendering system. Deals with setting up and shutting down
30 // the window + OpenGL context (based on SDL), and creating/using resources
31 // such as shaders, textures, and geometry.
32 class Renderer {
33  public:
34  // Creates the window + OpenGL context.
35  // A descriptive error is in last_error() if it returns false.
36  bool Initialize(const vec2i &window_size = vec2i(800, 600),
37  const char *window_title = "");
38 
39  // Swaps frames. Call this once per frame inside your main loop.
40  void AdvanceFrame(bool minimized);
41 
42  // Cleans up whatever Initialize creates.
43  void ShutDown();
44 
45  // Clears the framebuffer. Call this after AdvanceFrame if desired.
46  void ClearFrameBuffer(const vec4 &color);
47 
48  // Create a shader object from two strings containing glsl code.
49  // Returns nullptr upon error, with a descriptive message in glsl_error().
50  // Attribute names in the vertex shader should be aPosition, aNormal,
51  // aTexCoord and aColor to match whatever attributes your vertex data has.
52  Shader *CompileAndLinkShader(const char *vs_source, const char *ps_source);
53 
54  // Create a texture from a memory buffer containing xsize * ysize RGBA pixels.
55  // Return 0 if not a power of two in size.
56  GLuint CreateTexture(const uint8_t *buffer, const vec2i &size, bool has_alpha,
57  TextureFormat desired = kFormatAuto);
58 
59  // Unpacks a memory buffer containing a TGA format file.
60  // May only be uncompressed RGB or RGBA data, Y-flipped or not.
61  // Returns RGBA array of returned dimensions or nullptr if the
62  // format is not understood.
63  // You must free() the returned pointer when done.
64  uint8_t *UnpackTGA(const void *tga_buf, vec2i *dimensions, bool *has_alpha);
65 
66  // Unpacks a memory buffer containing a Webp format file.
67  // Returns RGBA array of the returned dimensions or nullptr if the format
68  // is not understood.
69  // You must free() the returned pointer when done.
70  uint8_t *UnpackWebP(const void *webp_buf, size_t size, vec2i *dimensions,
71  bool *has_alpha);
72 
73  // Loads the file in filename, and then unpacks the file format (supports
74  // TGA and WebP).
75  // last_error() contains more information if nullptr is returned.
76  // You must free() the returned pointer when done.
77  uint8_t *LoadAndUnpackTexture(const char *filename, vec2i *dimensions,
78  bool *has_alpha);
79 
80  // Utility functions to convert 32bit RGBA to 16bit.
81  // You must delete[] the return value afterwards.
82  uint16_t *Convert8888To5551(const uint8_t *buffer, const vec2i &size);
83  uint16_t *Convert888To565(const uint8_t *buffer, const vec2i &size);
84 
85  // Set alpha test (cull pixels with alpha below amount) vs alpha blend
86  // (blend with framebuffer pixel regardedless).
87  // blend_mode: see materials.fbs for valid enum values.
88  void SetBlendMode(BlendMode blend_mode, float amount = 0.5f);
89 
90  // Set to compare fragment against Z-buffer before writing, or not.
91  void DepthTest(bool on);
92 
93  // Call before rendering for Cardboard to set up the framebuffer
94  void BeginUndistortFramebuffer();
95  // Call when finished with Cardboard, to undistort and render the framebuffer
96  void FinishUndistortFramebuffer();
97 
98  Renderer()
99  : model_view_projection_(mat4::Identity()),
100  model_(mat4::Identity()),
101  color_(mathfu::kOnes4f),
102  light_pos_(mathfu::kZeros3f),
103  camera_pos_(mathfu::kZeros3f),
104  window_size_(mathfu::kZeros2i),
105  window_(nullptr),
106  context_(nullptr),
107  undistortFramebufferId_(0),
108  undistortTextureId_(0),
109  undistortRenderbufferId_(0) {}
110  ~Renderer() { ShutDown(); }
111 
112  // Shader uniform: model_view_projection
113  mat4 &model_view_projection() { return model_view_projection_; }
114  const mat4 &model_view_projection() const { return model_view_projection_; }
115 
116  // Shader uniform: model (object to world transform only)
117  mat4 &model() { return model_; }
118  const mat4 &model() const { return model_; }
119 
120  // Shader uniform: color
121  vec4 &color() { return color_; }
122  const vec4 &color() const { return color_; }
123 
124  // Shader uniform: light_pos
125  vec3 &light_pos() { return light_pos_; }
126  const vec3 &light_pos() const { return light_pos_; }
127 
128  // Shader uniform: camera_pos
129  vec3 &camera_pos() { return camera_pos_; }
130  const vec3 &camera_pos() const { return camera_pos_; }
131 
132  // If any of the more complex loading operations (shaders, textures etc.)
133  // fail, this sting will contain a more informative error message.
134  std::string &last_error() { return last_error_; }
135  const std::string &last_error() const { return last_error_; }
136 
137  // The device's current framebuffer size. May change from frame to frame
138  // due to window resizing or Android navigation buttons turning on/off.
139  vec2i &window_size() { return window_size_; }
140  const vec2i &window_size() const { return window_size_; }
141 
142  private:
143  GLuint CompileShader(GLenum stage, GLuint program, const GLchar *source);
144 
145  // Initializes the framebuffer needed for Cardboard mode
146  void InitializeUndistortFramebuffer(int width, int height);
147 
148  // The mvp. Use the Ortho() and Perspective() methods in mathfu::Matrix
149  // to conveniently change the camera.
150  mat4 model_view_projection_;
151  mat4 model_;
152  vec4 color_;
153  vec3 light_pos_;
154  vec3 camera_pos_;
155 
156  vec2i window_size_;
157 
158  std::string last_error_;
159 
160  SDL_Window *window_;
161  SDL_GLContext context_;
162 
163  BlendMode blend_mode_;
164 
165  bool use_16bpp_;
166 
167  // The id of the framebuffer that is used for rendering for Cardboard.
168  // After rendering to it, passed to Cardboard's undistortTexture call, which
169  // will transform and render it appropriately
170  GLuint undistortFramebufferId_;
171  // The texture that is used with the framebuffer, needed for the color
172  GLuint undistortTextureId_;
173  // The renderbuffer that is used with the framebuffer, needed for the depth
174  GLuint undistortRenderbufferId_;
175 };
176 
177 } // namespace fpl
